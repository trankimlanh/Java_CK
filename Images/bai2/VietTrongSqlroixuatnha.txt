CREATE TABLE DONGVAT (
    TenDongVat VARCHAR(10) PRIMARY KEY,
    Loai VARCHAR(10) CHECK (Loai IN ('Bo', 'Meo', 'Tho'))
);

CREATE TABLE THUCAN (
    TenThucAn VARCHAR(10) PRIMARY KEY,
    Loai VARCHAR(20) CHECK (Loai IN ('Co', 'Thit', 'Carot'))
);


CREATE TABLE AN (
    TenDongVat VARCHAR(10),
    TenThucAn VARCHAR(10),
    CoTheAnDuoc VARCHAR(10),
    PRIMARY KEY (TenDongVat, TenThucAn),
    FOREIGN KEY (TenDongVat) REFERENCES DONGVAT(TenDongVat),
    FOREIGN KEY (TenThucAn) REFERENCES THUCAN(TenThucAn)
);
